48 for(int j = 0; j < m; ++j) // Reconstruct slopes
49 { /*
50 * j-1 j j+1

53 */
54 if(j)

64 s_L = (U[j][i] - UL) / h;
65 }
66 if(j < m-1)

68 if (NO_h)

70 s_R = (U[j+1][i] - U[j][i]) / h;

72 else

77 }
