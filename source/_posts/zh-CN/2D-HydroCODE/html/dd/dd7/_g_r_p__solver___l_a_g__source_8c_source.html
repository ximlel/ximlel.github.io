54 * \frac{\partial [ifv_L.RHO, u, p, ifv_R.RHO]}{\partial t}

56 * [rho_star_L, u_star, p_star, rho_star_R]

62 double ** P = CV.P;

64 // the slopes of variable values
65 double * s_rho = calloc(m, sizeof(double));
66 double * s_u = calloc(m, sizeof(double));
67 double * s_p = calloc(m, sizeof(double));
68 CV.d_rho = s_rho;
69 CV.d_u = s_u;
70 CV.d_p = s_p;
71 // the variable values at (x_{j-1/2}, t_{n+1}).
72 double * U_next = malloc((m+1) * sizeof(double));
73 double * P_next = malloc((m+1) * sizeof(double));
74 double * RHO_next_L = malloc((m+1) * sizeof(double));
75 double * RHO_next_R = malloc((m+1) * sizeof(double));
76 // the temporal derivatives at (x_{j-1/2}, t_{n}).
77 double * U_t = malloc((m+1) * sizeof(double));

79 double * RHO_t_L = malloc((m+1) * sizeof(double));

81 // the numerical flux at (x_{j-1/2}, t_{n+1/2}).
82 double * U_F = malloc((m+1) * sizeof(double));

84 double * MASS = malloc(m * sizeof(double)); // Array of the mass data in computational cells.
85 if(s_rho == NULL || s_u == NULL || s_p == NULL)

87 printf("NOT enough memory! Slope\n");

113 struct b_f_var bfv_L = {.H = h, .SU = 0.0, .SP = 0.0, .SRHO = 0.0}; // Left boundary condition
114 struct b_f_var bfv_R = {.H = h, .SU = 0.0, .SP = 0.0, .SRHO = 0.0}; // Right boundary condition
115 struct i_f_var ifv_L = {.gamma = gamma}, ifv_R = {.gamma = gamma};

117//-----------------------THE MAIN LOOP--------------------------------
118 for(k = 1; k <= N; ++k)
119 {
120 h_S_max = INFINITY; // h/S_max = INFINITY
121 tic = clock();
122
123 find_bound = bound_cond_slope_limiter(true, m, nt-1, CV, &bfv_L, &bfv_R, find_bound, true, time_c, X[nt-1]);
124 if(!find_bound)
125 goto return_NULL;

127 for(j = 0; j <= m; ++j)
128 { /*
129 * j-1 j j+1
130 * j-1/2 j-1 j+1/2 j j+3/2 j+1
131 * o-----X-----o-----X-----o-----X--...
132 */
133 if(j) // Initialize the initial values.
134 {
135 h_L = X[nt-1][j] - X[nt-1][j-1];
136 ifv_L.RHO = RHO[nt-1][j-1] + 0.5*h_L*s_rho[j-1];
137 ifv_L.U = U[nt-1][j-1] + 0.5*h_L*s_u[j-1];
138 ifv_L.P = P[nt-1][j-1] + 0.5*h_L*s_p[j-1];
139 }
140 else
141 {
142 h_L = bfv_L.H;
143 ifv_L.RHO = bfv_L.RHO + 0.5*h_L*bfv_L.SRHO;
144 ifv_L.U = bfv_L.U + 0.5*h_L*bfv_L.SU;
145 ifv_L.P = bfv_L.P + 0.5*h_L*bfv_L.SP;
146 }
147 if(j < m)
148 {
149 h_R = X[nt-1][j+1] - X[nt-1][j];
150 ifv_R.RHO = RHO[nt-1][j] - 0.5*h_R*s_rho[j];
151 ifv_R.U = U[nt-1][j] - 0.5*h_R*s_u[j];
152 ifv_R.P = P[nt-1][j] - 0.5*h_R*s_p[j];
153 }
154 else
155 {
156 h_R = bfv_R.H;
157 ifv_R.RHO = bfv_R.RHO + 0.5*h_R*bfv_R.SRHO;
158 ifv_R.U = bfv_R.U + 0.5*h_R*bfv_R.SU;
159 ifv_R.P = bfv_R.P + 0.5*h_R*bfv_R.SP;
160 }
161 if(ifv_L.P < eps || ifv_R.P < eps || ifv_L.RHO < eps || ifv_R.RHO < eps)
162 {
163 printf("<0.0 error on [%d, %d] (t_n, x) - Reconstruction\n", k, j);
164 goto return_NULL;
165 }
166
167 c_L = sqrt(gamma * ifv_L.P / ifv_L.RHO);
168 c_R = sqrt(gamma * ifv_R.P / ifv_R.RHO);
169 h_S_max = fmin(h_S_max, h_L/c_L);
170 h_S_max = fmin(h_S_max, h_R/c_R);
171 if ((bound == -2 || bound == -24) && j == 0) // reflective boundary conditions
172 h_S_max = fmin(h_S_max, h_L/(fabs(ifv_L.U)+c_L));
173 if (bound == -2 && j == m)
174 h_S_max = fmin(h_S_max, h_R/(fabs(ifv_R.U)+c_R));
175
176 if(j) //calculate the material derivatives
177 {
178 ifv_L.t_u = s_u[j-1]/ifv_L.RHO;
179 ifv_L.t_p = s_p[j-1]/ifv_L.RHO;
180 ifv_L.t_rho = s_rho[j-1]/ifv_L.RHO;
181 }
182 else
183 {
184 ifv_L.t_rho = bfv_L.SRHO/ifv_L.RHO;
185 ifv_L.t_u = bfv_L.SU /ifv_L.RHO;
186 ifv_L.t_p = bfv_L.SP /ifv_L.RHO;
187 }
188 if(j < m)
189 {
190 ifv_R.t_u = s_u[j]/ifv_R.RHO;
191 ifv_R.t_p = s_p[j]/ifv_R.RHO;
192 ifv_R.t_rho = s_rho[j]/ifv_R.RHO;
193 }
194 else
195 {
196 ifv_R.t_rho = bfv_R.SRHO/ifv_R.RHO;
197 ifv_R.t_u = bfv_R.SU /ifv_R.RHO;
198 ifv_R.t_p = bfv_R.SP /ifv_R.RHO;
199 }
200 if(!isfinite(ifv_L.t_p)|| !isfinite(ifv_R.t_p)|| !isfinite(ifv_L.t_u)|| !isfinite(ifv_R.t_u)|| !isfinite(ifv_L.t_rho)|| !isfinite(ifv_R.t_rho))

202 printf("NAN or INFinite error on [%d, %d] (t_n, x) - Slope\n", k, j);

209 if(mid[2] < eps || mid[0] < eps || mid[3] < eps)
210 {
211 printf("<0.0 error on [%d, %d] (t_n, x) - STAR\n", k, j);
212 time_c = t_all;
213 }
214 if(!isfinite(mid[1])|| !isfinite(mid[2])|| !isfinite(mid[0])|| !isfinite(mid[3]))
215 {
216 printf("NAN or INFinite error on [%d, %d] (t_n, x) - STAR\n", k, j);
217 time_c = t_all;
218 }
219 if(!isfinite(dire[1])|| !isfinite(dire[2])|| !isfinite(dire[0])|| !isfinite(dire[3]))
220 {
221 printf("NAN or INFinite error on [%d, %d] (t_n, x) - DIRE\n", k, j);
222 time_c = t_all;
223 }

226 RHO_next_R[j] = mid[3];
227 U_next[j] = mid[1];
228 P_next[j] = mid[2];
229 RHO_t_L[j] = dire[0];
230 RHO_t_R[j] = dire[3];
231 U_t[j] = dire[1];
232 P_t[j] = dire[2];
233 }
234
235//====================Time step and grid movement======================
236 // If no total time, use fixed tau and time step N.
237 if (isfinite(t_all) || !isfinite(config[16]) || config[16] <= 0.0)
238 {
239 tau = fmin(CFL * h_S_max, C_m * tau);
240 if ((time_c + tau) > (t_all - eps))
241 tau = t_all - time_c;
242 else if(!isfinite(tau))
243 {
244 printf("NAN or INFinite error on [%d, %g] (t_n, tau) - CFL\n", k, tau);
245 tau = t_all - time_c;
246 goto return_NULL;
247 }
248 }
249
250 for(j = 0; j <= m; ++j)
251 {
252 U_F[j] = U_next[j] + 0.5 * tau * U_t[j];
253 P_F[j] = P_next[j] + 0.5 * tau * P_t[j];
254
255 RHO_next_L[j] += tau * RHO_t_L[j];
256 RHO_next_R[j] += tau * RHO_t_R[j];
257 U_next[j] += tau * U_t[j];
258 P_next[j] += tau * P_t[j];
259
260 X[nt][j] = X[nt-1][j] + tau * U_F[j]; // motion along the contact discontinuity
261 }
262
263//======================THE CORE ITERATION=========================(On Lagrangian Coordinate)
264 for(j = 0; j < m; ++j) // forward Euler
265 { /*
266 * j-1 j j+1
267 * j-1/2 j-1 j+1/2 j j+3/2 j+1
268 * o-----X-----o-----X-----o-----X--...
269 */
270 RHO[nt][j] = 1.0 / (1.0/RHO[nt-1][j] + tau/MASS[j]*(U_F[j+1] - U_F[j]));
271 U[nt][j] = U[nt-1][j] - tau/MASS[j]*(P_F[j+1] - P_F[j]);
272 E[nt][j] = E[nt-1][j] - tau/MASS[j]*(P_F[j+1]*U_F[j+1] - P_F[j]*U_F[j]);
273 P[nt][j] = (E[nt][j] - 0.5 * U[nt][j]*U[nt][j]) * (gamma - 1.0) * RHO[nt][j];
274 if(P[nt][j] < eps || RHO[nt][j] < eps)
275 {
276 printf("<0.0 error on [%d, %d] (t_n, x) - Update\n", k, j);
277 time_c = t_all;
278 }
279
280//============================compute the slopes============================
281 s_u[j] = ( U_next[j+1] - U_next[j])/(X[nt][j+1]-X[nt][j]);
282 s_p[j] = ( P_next[j+1] - P_next[j])/(X[nt][j+1]-X[nt][j]);
283 s_rho[j] = (RHO_next_L[j+1] - RHO_next_R[j])/(X[nt][j+1]-X[nt][j]);
284 }
285
286//============================Time update=======================
287
288 toc = clock();
289 cpu_time[nt] = ((double)toc - (double)tic) / (double)CLOCKS_PER_SEC;;
290 cpu_time_sum += cpu_time[nt];
291
292 time_c += tau;
293 if (isfinite(t_all))
294 DispPro(time_c*100.0/t_all, k);
295 else
296 DispPro(k*100.0/N, k);
297 if(time_c > (t_all - eps) || isinf(time_c))
298 {
299 config[5] = (double)k;
300 break;
301 }
302
303//===========================Fixed variable location=======================
304 for(j = 0; j <= m; ++j)
305 X[nt-1][j] = X[nt][j];
306 for(j = 0; j < m; ++j)
307 {
308 RHO[nt-1][j] = RHO[nt][j];
309 U[nt-1][j] = U[nt][j];
310 E[nt-1][j] = E[nt][j];
311 P[nt-1][j] = P[nt][j];
312 }
313 }
314
315 time_plot[0] = time_c - tau;
316 time_plot[1] = time_c;
317 printf("\nTime is up at time step %d.\n", k);
318 printf("The cost of CPU time for 1D-GRP Lagrangian scheme for this problem is %g seconds.\n", cpu_time_sum);
319//---------------------END OF THE MAIN LOOP----------------------
320
321return_NULL:
322 free(s_u);
323 free(s_p);
324 free(s_rho);
325 s_u = NULL;
326 s_p = NULL;
327 s_rho = NULL;
328 free(U_next);
329 free(P_next);
330 free(RHO_next_L);
331 free(RHO_next_R);
332 U_next = NULL;
333 P_next = NULL;
334 RHO_next_L = NULL;
335 RHO_next_R = NULL;
336 free(U_t);
337 free(P_t);
338 free(RHO_t_L);
339 free(RHO_t_R);
340 U_t = NULL;
341 P_t = NULL;
342 RHO_t_L = NULL;
343 RHO_t_R = NULL;
344 free(U_F);
345 free(P_F);
346 U_F = NULL;
347 P_F = NULL;
348 free(MASS);
349 MASS = NULL;
350}
