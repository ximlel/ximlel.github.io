66 * i is a frequently used index for y-spatial variables.
67 * j is a frequently used index for x-spatial variables.
68 * k is a frequently used index for the time step.
69 */
70 int i, j, k;

73 double cpu_time_sum = 0.0;

75 double const t_all = config[1]; // the total time
76 double const eps = config[4]; // the largest value could be seen as zero
77 int const N = (int)(config[5]); // the maximum number of time steps
78 double const gamma = config[6]; // the constant of the perfect gas
79 double const CFL = config[7]; // the CFL number
80 double const h_x = config[10]; // the length of the initial x-spatial grids
81 double const h_y = config[11]; // the length of the initial y-spatial grids
82 double tau = config[16]; // the length of the time step
83
84 _Bool find_bound_x = false, find_bound_y = false;
85 int flux_err;

87 double mom_x, mom_y, ene;
88 double c; // the speeds of sound

90 // Left/Right/Upper/Downside boundary condition
91 struct b_f_var * bfv_L = NULL, * bfv_R = NULL, * bfv_U = NULL, * bfv_D = NULL;
92 // the slopes of variable values.
93 _2D_INIT_MEM(s_rho, m, n); _2D_INIT_MEM(t_rho, m, n);
94 _2D_INIT_MEM(s_u, m, n); _2D_INIT_MEM(t_u, m, n);
95 _2D_INIT_MEM(s_v, m, n); _2D_INIT_MEM(t_v, m, n);
96 _2D_INIT_MEM(s_p, m, n); _2D_INIT_MEM(t_p, m, n);
97 // the variable values at (x_{j-1/2}, t_{n+1}).
98 _2D_INIT_MEM(rhoIx, m+1, n);
99 _2D_INIT_MEM(uIx, m+1, n);
100 _2D_INIT_MEM(vIx, m+1, n);
101 _2D_INIT_MEM(pIx, m+1, n);
102 _2D_INIT_MEM(F_rho, m+1, n);
103 _2D_INIT_MEM(F_u, m+1, n);
104 _2D_INIT_MEM(F_v, m+1, n);
105 _2D_INIT_MEM(F_e, m+1, n);
106 // the variable values at (y_{j-1/2}, t_{n+1}).
107 _2D_INIT_MEM(rhoIy, m, n+1);
108 _2D_INIT_MEM(uIy, m, n+1);
109 _2D_INIT_MEM(vIy, m, n+1);
110 _2D_INIT_MEM(pIy, m, n+1);
111 _2D_INIT_MEM(G_rho, m, n+1);
112 _2D_INIT_MEM(G_u, m, n+1);
113 _2D_INIT_MEM(G_v, m, n+1);
114 _2D_INIT_MEM(G_e, m, n+1);
115 // boundary condition
116 _1D_BC_INIT_MEM(bfv_L, n); _1D_BC_INIT_MEM(bfv_R, n);
117 _1D_BC_INIT_MEM(bfv_D, m); _1D_BC_INIT_MEM(bfv_U, m);
118
119 double mu, nu; // nu = tau/h_x, mu = tau/h_y.
120
121 double h_S_max, sigma; // h/S_max, S_max is the maximum character speed, sigma is the character speed
122 double time_c = 0.0; // the current time
123 int nt = 1; // the number of times storing plotting data
124
125//------------THE MAIN LOOP-------------
126 for(k = 1; k <= N; ++k)
127 {
128 /* evaluate f and a at some grid points for the iteration
129 * and evaluate the character speed to decide the length
130 * of the time step by (tau * speed_max)/h = CFL
131 */
132 h_S_max = INFINITY; // h/S_max = INFINITY
133 tic = clock();
134
135 for(j = 0; j < m; ++j)
136 for(i = 0; i < n; ++i)
137 {
138 c = sqrt(gamma * CV->P[j][i] / CV->RHO[j][i]);
139 sigma = fabs(c) + fabs(CV->U[j][i]) + fabs(CV->V[j][i]);
140 h_S_max = fmin(h_S_max, fmin(h_x,h_y) / sigma);
141 }
142 // If no total time, use fixed tau and time step N.
143 if (isfinite(t_all) || !isfinite(config[16]) || config[16] <= 0.0)
144 {
145 tau = CFL * h_S_max;
146 if ((time_c + tau) > (t_all - eps))
147 tau = t_all - time_c;
148 else if(!isfinite(tau))
149 {
150 printf("NAN or INFinite error on [%d, %g] (t_n, tau) - CFL\n", k, tau);
151 tau = t_all - time_c;
152 goto return_NULL;
153 }
154 }
155 nu = tau / h_x;
156 mu = tau / h_y;
157
158
159 find_bound_x = bound_cond_slope_limiter_x(m, n, nt-1, CV, bfv_L, bfv_R, bfv_D, bfv_U, find_bound_x, true, time_c);
160 if(!find_bound_x)
161 goto return_NULL;
162 find_bound_y = bound_cond_slope_limiter_y(m, n, nt-1, CV, bfv_L, bfv_R, bfv_D, bfv_U, find_bound_y, true, time_c);
163 if(!find_bound_y)
164 goto return_NULL;
165
166 flux_err = flux_generator_x(m, n, nt-1, tau, CV, bfv_L, bfv_R, true);
167 if(flux_err == 1)
168 goto return_NULL;
169 else if(flux_err == 2)
170 time_c = t_all;
171 flux_err = flux_generator_y(m, n, nt-1, tau, CV, bfv_D, bfv_U, true);
172 if(flux_err == 1)
173 goto return_NULL;
174 else if(flux_err == 2)
175 time_c = t_all;
176
177//===============THE CORE ITERATION=================
178 for(i = 0; i < n; ++i)
179 for(j = 0; j < m; ++j)
180 { /*
181 * j-1 j j+1
182 * j-1/2 j-1 j+1/2 j j+3/2 j+1
183 * o-----X-----o-----X-----o-----X--...
184 */
185 CV[nt].RHO[j][i] = CV[nt-1].RHO[j][i] - nu*(CV->F_rho[j+1][i]-CV->F_rho[j][i]) - mu*(CV->G_rho[j][i+1]-CV->G_rho[j][i]);
186 mom_x = CV[nt-1].RHO[j][i]*CV[nt-1].U[j][i] - nu*(CV->F_u[j+1][i] -CV->F_u[j][i]) - mu*(CV->G_u[j][i+1] -CV->G_u[j][i]);
187 mom_y = CV[nt-1].RHO[j][i]*CV[nt-1].V[j][i] - nu*(CV->F_v[j+1][i] -CV->F_v[j][i]) - mu*(CV->G_v[j][i+1] -CV->G_v[j][i]);
188 ene = CV[nt-1].RHO[j][i]*CV[nt-1].E[j][i] - nu*(CV->F_e[j+1][i] -CV->F_e[j][i]) - mu*(CV->G_e[j][i+1] -CV->G_e[j][i]);
189
190 CV[nt].U[j][i] = mom_x / CV[nt].RHO[j][i];
191 CV[nt].V[j][i] = mom_y / CV[nt].RHO[j][i];
192 CV[nt].E[j][i] = ene / CV[nt].RHO[j][i];
193 CV[nt].P[j][i] = (ene - 0.5*mom_x*CV[nt].U[j][i] - 0.5*mom_y*CV[nt].V[j][i])*(gamma-1.0);
194
195 CV->s_rho[j][i] = (CV->rhoIx[j+1][i] - CV->rhoIx[j][i])/h_x;
196 CV->s_u[j][i] = ( CV->uIx[j+1][i] - CV->uIx[j][i])/h_x;
197 CV->s_v[j][i] = ( CV->vIx[j+1][i] - CV->vIx[j][i])/h_x;
198 CV->s_p[j][i] = ( CV->pIx[j+1][i] - CV->pIx[j][i])/h_x;
199 CV->t_rho[j][i] = (CV->rhoIy[j][i+1] - CV->rhoIy[j][i])/h_y;
200 CV->t_u[j][i] = ( CV->uIy[j][i+1] - CV->uIy[j][i])/h_y;
201 CV->t_v[j][i] = ( CV->vIy[j][i+1] - CV->vIy[j][i])/h_y;
202 CV->t_p[j][i] = ( CV->pIy[j][i+1] - CV->pIy[j][i])/h_y;
203 }
204
205//==================================================
206
207 toc = clock();
208 cpu_time[nt] = ((double)toc - (double)tic) / (double)CLOCKS_PER_SEC;;
209 cpu_time_sum += cpu_time[nt];
210
211 time_c += tau;
212 if (isfinite(t_all))
213 DispPro(time_c*100.0/t_all, k);
214 else
215 DispPro(k*100.0/N, k);
216 if(time_c > (t_all - eps) || isinf(time_c))
217 {
218 config[5] = (double)k;
219 break;
220 }
221
222 //===========================Fixed variable location=======================
223 for(j = 0; j < m; ++j)
224 for(i = 0; i < n; ++i)
225 {
226 CV[nt-1].RHO[j][i] = CV[nt].RHO[j][i];
227 CV[nt-1].U[j][i] = CV[nt].U[j][i];
228 CV[nt-1].V[j][i] = CV[nt].V[j][i];
229 CV[nt-1].E[j][i] = CV[nt].E[j][i];
230 CV[nt-1].P[j][i] = CV[nt].P[j][i];
231 }
232 }
233
234 time_plot[0] = time_c - tau;
235 time_plot[1] = time_c;
236 printf("\nTime is up at time step %d.\n", k);
237 printf("The cost of CPU time for genuinely 2D-GRP Eulerian scheme without dimension splitting for this problem is %g seconds.\n", cpu_time_sum);
238 //------------END OF THE MAIN LOOP-------------
239
240return_NULL:

242 {

246 CV->rhoIx[j]= NULL; CV->uIx[j]= NULL; CV->vIx[j]= NULL; CV->pIx[j]= NULL;
247 }

253 free(CV->t_rho[j]); free(CV->t_u[j]); free(CV->t_v[j]); free(CV->t_p[j]);
254
255 CV->G_rho[j]= NULL; CV->G_u[j]= NULL; CV->G_v[j]= NULL; CV->G_e[j]= NULL;
256 CV->rhoIy[j]= NULL; CV->uIy[j]= NULL; CV->vIy[j]= NULL; CV->pIy[j]= NULL;
257 CV->s_rho[j]= NULL; CV->s_u[j]= NULL; CV->s_v[j]= NULL; CV->s_p[j]= NULL;
258 CV->t_rho[j]= NULL; CV->t_u[j]= NULL; CV->t_v[j]= NULL; CV->t_p[j]= NULL;
259 }
260 free(CV->F_rho); free(CV->F_u); free(CV->F_v); free(CV->F_e);
261 free(CV->rhoIx); free(CV->uIx); free(CV->vIx); free(CV->pIx);
262 free(CV->G_rho); free(CV->G_u); free(CV->G_v); free(CV->G_e);
263 free(CV->rhoIy); free(CV->uIy); free(CV->vIy); free(CV->pIy);
264 free(CV->s_rho); free(CV->s_u); free(CV->s_v); free(CV->s_p);
265 free(CV->t_rho); free(CV->t_u); free(CV->t_v); free(CV->t_p);
266 free(bfv_L); free(bfv_R);
267 free(bfv_D); free(bfv_U);
268
269 CV->F_rho= NULL; CV->F_u= NULL; CV->F_v= NULL; CV->F_e= NULL;
270 CV->rhoIx= NULL; CV->uIx= NULL; CV->vIx= NULL; CV->pIx= NULL;
271 CV->G_rho= NULL; CV->G_u= NULL; CV->G_v= NULL; CV->G_e= NULL;
272 CV->rhoIy= NULL; CV->uIy= NULL; CV->vIy= NULL; CV->pIy= NULL;
273 CV->s_rho= NULL; CV->s_u= NULL; CV->s_v= NULL; CV->s_p= NULL;
274 CV->t_rho= NULL; CV->t_u= NULL; CV->t_v= NULL; CV->t_p= NULL;
275 bfv_L= NULL; bfv_R= NULL;
