422 a_R = 1.0 +rho_star_R* SmUs * H1;
423 b_R = -(SmUs/(rho_star_R*c_star_R*c_star_R) + H1);
424 d_R = L_rho*d_rho_R + L_u*d_u_R + L_p*d_p_R;
425 }
426
427 detA = a_L*b_R - b_L*a_R;
428 u_t_mat = (b_R*d_L - b_L*d_R)/detA;
429 p_t_mat = (a_L*d_R - a_R*d_L)/detA;
430
431 //already total D!
432 D[1] = u_t_mat + (u_star-lambda_u)/U[0]/C/C * p_t_mat;
433 D[3] = p_t_mat + (u_star-lambda_u)*U[0] * u_t_mat;
434
435 if(u_star < lambda_u) //the direction is between the contact discontinuety and the 3-wave
436 {
437 if(CRW[1]) //the 3-wave is a CRW
438 {
439 //already total D!
440 D[0] = rho_star_R*(u_star-lambda_u)*pow(c_star_R/c_R, (1.0+zetaR)/zetaR)*(d_p_R - d_rho_R*c_R*c_R)/rho_R;
441 D[0] = (D[0] + D[3]) / c_star_R/c_star_R;
442
443 D[2] = -U[1]*d_v_R*U[0]/rho_R;
444 D[2] = D[2] + lambda_u*d_v_R;
445 D[4] = -U[1]*d_z_R*U[0]/rho_R;
446 D[4] = D[4] + lambda_u*d_z_R;
447 D[5] = -U[1]*d_phi_R*U[0]/rho_R;
448 D[5] = D[5] + lambda_u*d_phi_R;
449 }
450 else //the 3-wave is a shock
451 {
452 SmUs = sqrt(0.5*((gammaR+1.0)*p_R + (gammaR-1.0)*p_star)/rho_star_R);
453 SmUR = sqrt(0.5*((gammaR+1.0)*p_star+ (gammaR-1.0)*p_R )/rho_R);
454
455 VAR = p_R + zetaR*p_star;
456 H1 = rho_R * p_R * (1.0 - zetaR*zetaR) / VAR/VAR;
457 H2 = rho_R * p_star * (zetaR*zetaR - 1.0) / VAR/VAR;
458 H3 = (p_star + zetaR*p_R)/VAR;
459
460 L_rho = SmUR * H3 * d_rho_R;
461 L_u = -rho_R * (H2*c_R*c_R + H3) * d_u_R;
462 L_p = H2 * SmUR * d_p_R;
463
464 D[0] = ((u_star+SmUs)/c_star_R/c_star_R - u_star*H1)*p_t_mat + rho_star_R*u_star*SmUs*H1*u_t_mat;
465 D[0] = (D[0] - u_star*(L_p+L_rho+L_u)) / SmUs;
466
467 f = SmUR*(H2*d_p_R + H3*d_rho_R) - rho_R*(H2*c_R*c_R+H3)*d_u_R;
468 rho_x = (f + H1*(p_t_mat - rho_star_R*SmUs*u_t_mat) - D[0]) / (SmUR+u_R);//shk_spd;
469 D[0] = D[0] + lambda_u*rho_x;
470
471 D[2] = -U[1] * SmUR * d_v_R / SmUs;
472 D[2] = D[2] + lambda_u*d_v_R;
473 D[4] = -U[1] * SmUR * d_z_R / SmUs;
474 D[4] = D[4] + lambda_u*d_z_R;
475 D[5] = -U[1] * SmUR * d_phi_R / SmUs;
476 D[5] = D[5] + lambda_u*d_phi_R;
477 }
478 }
479 else //the direction is between the 1-wave and the contact discontinuety
480 {
481 if(CRW[0]) //the 1-wave is a CRW
482 {
483 //already total D!
484 D[0] = rho_star_L*(u_star-lambda_u)*pow(c_star_L/c_L, (1.0+zetaL)/zetaL)*(d_p_L - d_rho_L*c_L*c_L)/rho_L;
485 D[0] = (D[0] + D[3]) / c_star_L/c_star_L;
486
487 D[2] = -U[1]*d_v_L*U[0]/rho_L;
488 D[2] = D[2] + lambda_u*d_v_L;
489 D[4] = -U[1]*d_z_L*U[0]/rho_L;
490 D[4] = D[4] + lambda_u*d_z_L;
491 D[5] = -U[1]*d_phi_L*U[0]/rho_L;
492 D[5] = D[5] + lambda_u*d_phi_L;
493 }
494 else //the 1-wave is a shock
495 {
496 SmUs = -sqrt(0.5*((gammaL+1.0)*p_L +(gammaL-1.0)*p_star)/rho_star_L);
497 SmUL = -sqrt(0.5*((gammaL+1.0)*p_star+(gammaL-1.0)*p_L )/rho_L);
498
499 VAR = p_L + zetaL*p_star;
500 H1 = rho_L * p_L * (1.0 - zetaL*zetaL) / VAR/VAR;
501 H2 = rho_L * p_star * (zetaL*zetaL - 1.0) / VAR/VAR;
502 H3 = (p_star + zetaL*p_L)/VAR;
503
504 L_rho = SmUL * H3 * d_rho_L;
505 L_u = -rho_L*(H2*c_L*c_L + H3) * d_u_L;
506 L_p = H2 * SmUL * d_p_L;
507
508 D[0] = ((u_star+SmUs)/c_star_L/c_star_L - H1*u_star)*p_t_mat + rho_star_L*u_star*SmUs*H1*u_t_mat;
509 D[0] = (D[0] - u_star*(L_p+L_rho+L_u))/ SmUs;
510
511 f = SmUL*(H2*d_p_L + H3*d_rho_L) - rho_L*(H2*c_L*c_L+H3)*d_u_L;
512 rho_x = (f + H1*(p_t_mat - rho_star_L*SmUs*u_t_mat) - D[0]) / (SmUL+u_L);
513 D[0] = D[0] + lambda_u*rho_x;
514
515 D[2] = -U[1] * SmUL * d_v_L / SmUs;
516 D[2] = D[2] + lambda_u*d_v_L;
517 D[4] = -U[1] * SmUL * d_z_L / SmUs;
518 D[4] = D[4] + lambda_u*d_z_L;
519 D[5] = -U[1] * SmUL * d_phi_L / SmUs;
520 D[5] = D[5] + lambda_u*d_phi_L;
521 }
522 }
523 //--end of non-sonic case--
524 }
525 T_p = 0.5*((t_u_L*(U[0]*C) + t_p_L) - (t_u_R*(U[0]*C) - t_p_R));
526 T_u = 0.5*(t_u_L + t_p_L/(U[0]*C) + t_u_R - t_p_R/(U[0]*C));
527 if (u_star > lambda_u)
528 {
529 T_rho = t_rho_L - t_p_L/(C*C) + T_p/(C*C);
530 D[0] = D[0] - (U[2]-lambda_v)*T_rho - U[0]*t_v_L;
531 D[1] = D[1] - (U[2]-lambda_v)*T_u;
532 D[2] = D[2] - (U[2]-lambda_v)*t_v_L - T_p/U[0];
533 D[3] = D[3] - (U[2]-lambda_v)*T_p - U[0]*C*C*t_v_L;
534 D[4] = D[4] - (U[2]-lambda_v)*t_z_L;
535 D[5] = D[5] - (U[2]-lambda_v)*t_phi_L;
536 }
537 else
538 {
539 T_rho = t_rho_R - t_p_R/(C*C) + T_p/(C*C);
540 D[0] = D[0] - (U[2]-lambda_v)*T_rho - U[0]*t_v_R;
541 D[1] = D[1] - (U[2]-lambda_v)*T_u;
542 D[2] = D[2] - (U[2]-lambda_v)*t_v_R - T_p/U[0];
543 D[3] = D[3] - (U[2]-lambda_v)*T_p - U[0]*C*C*t_v_R;
544 D[4] = D[4] - (U[2]-lambda_v)*t_z_R;
545 D[5] = D[5] - (U[2]-lambda_v)*t_phi_R;
546 }
547 //----end of non-trivial case----
548 }
549 U_star[0] = rho_star_L;
550 U_star[1] = u_star;
551 U_star[2] = rho_star_R;
552 U_star[3] = p_star;
553 U_star[4] = c_star_L;
554 U_star[5] = c_star_R;
555}
