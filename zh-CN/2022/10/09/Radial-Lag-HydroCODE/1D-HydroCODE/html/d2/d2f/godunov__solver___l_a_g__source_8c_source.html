103 for(j = 0; j <= m; ++j)
104 { /*
105 * j-1 j j+1

107 * o-----X-----o-----X-----o-----X--...
108 */

112 ifv_L.RHO = RHO[nt][j-1];
113 ifv_L.U = U[nt][j-1];
114 ifv_L.P = P[nt][j-1];

121 ifv_L.P = bfv_L.P;
122 }
123 if(j < m)
124 {
125 h_R = X[nt][j+1] - X[nt][j];
126 ifv_R.RHO = RHO[nt][j];
127 ifv_R.U = U[nt][j];
128 ifv_R.P = P[nt][j];
129 }

133 ifv_R.RHO = bfv_R.RHO;
134 ifv_R.U = bfv_R.U;
135 ifv_R.P = bfv_R.P;
136 }

138 c_L = sqrt(gamma * ifv_L.P / ifv_L.RHO);
139 c_R = sqrt(gamma * ifv_R.P / ifv_R.RHO);
140 h_S_max = fmin(h_S_max, h_L/c_L);
141 h_S_max = fmin(h_S_max, h_R/c_R);
142 if ((bound == -2 || bound == -24) && j == 0) // reflective boundary conditions
143 h_S_max = fmin(h_S_max, h_L/(fabs(ifv_L.U)+c_L));
144 if (bound == -2 && j == m)
145 h_S_max = fmin(h_S_max, h_R/(fabs(ifv_R.U)+c_R));

147//========================Solve Riemann Problem========================
148
149 Riemann_solver_exact_single(&u_star, &p_star, gamma, ifv_L.U, ifv_R.U, ifv_L.P, ifv_R.P, c_L, c_R, CRW, eps, eps, 500);

151 if(p_star < eps)

154 stop_t = true;
155 }
156 if(!isfinite(p_star)|| !isfinite(u_star))

166//====================Time step and grid movement======================
167 // If no total time, use fixed tau and time step N.
168 if(isfinite(t_all) || !isfinite(config[16]) || config[16] <= 0.0)
169 {
170 tau = fmin(CFL * h_S_max, C_m * tau);
171 if(tau < eps)
172 {
173 printf("\nThe length of the time step is so small on [%d, %g, %g] (t_n, time_c, tau)\n", k, time_c, tau);
174 stop_t = true;
175 }
176 else if((time_c + tau) > (t_all - eps))
177 tau = t_all - time_c;
178 else if(!isfinite(tau))

181 goto return_NULL;
182 }
183 }

185 for(j = 0; j <= m; ++j)
186 X[nt][j] += tau * U_F[j]; // motion along the contact discontinuity
187
188//======================THE CORE ITERATION=========================(On Lagrangian Coordinate)
189 for(j = 0; j < m; ++j) // forward Euler
190 { /*
191 * j-1 j j+1
192 * j-1/2 j-1 j+1/2 j j+3/2 j+1
193 * o-----X-----o-----X-----o-----X--...
194 */
195 RHO[nt][j] = 1.0 / (1.0/RHO[nt][j] + tau/MASS[j]*(U_F[j+1] - U_F[j]));
196 U[nt][j] = U[nt][j] - tau/MASS[j]*(P_F[j+1] - P_F[j]);
197 E[nt][j] = E[nt][j] - tau/MASS[j]*(P_F[j+1]*U_F[j+1] - P_F[j]*U_F[j]);
198 P[nt][j] = (E[nt][j] - 0.5 * U[nt][j]*U[nt][j]) * (gamma - 1.0) * RHO[nt][j];
199 if(P[nt][j] < eps || RHO[nt][j] < eps)
200 {
201 printf("<0.0 error on [%d, %d] (t_n, x) - Update\n", k, j);
202 stop_t = true;
203 }
204 }
205
206//============================Time update=======================
207
208 time_c += tau;
209 if(isfinite(t_all))
210 DispPro(time_c*100.0/t_all, k);
211 else
212 DispPro(k*100.0/N, k);
213 if(stop_t || time_c > (t_all - eps) || !isfinite(time_c))
214 break;
215
216//===========================Fixed variable location=======================
217
218 toc = clock();
219 cpu_time_sum += ((double)toc - (double)tic) / (double)CLOCKS_PER_SEC;
220 cpu_time[nt] = cpu_time_sum;
221 }
222
223 printf("\nTime is up at time step %d.\n", k);
224 printf("The cost of CPU time for 1D-Godunov Lagrangian scheme for this problem is %g seconds.\n", cpu_time_sum);
225//---------------------END OF THE MAIN LOOP----------------------
226
227return_NULL:
228 config[5] = (double)k;
229 *N_plot = nt+1;
230 if(isfinite(time_c))
231 time_plot[nt] = time_c;
232 else if(isfinite(t_all))
233 time_plot[nt] = t_all;
234 else if(isfinite(tau))
235 time_plot[nt] = k*tau;
236
237 free(U_F);
238 free(P_F);
239 U_F = NULL;
240 P_F = NULL;
241 free(MASS);
242 MASS = NULL;
