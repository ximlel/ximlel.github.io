28{
29 /*
30 * j is a frequently used index for spatial variables.
31 * k is a frequently used index for the time step.
32 */
33 int j, k = 0;
34
35 clock_t tic, toc;
36 double cpu_time_sum = 0.0;
37
38 double const t_all = config[1]; // the total time
39 double const eps = config[4]; // the largest value could be seen as zero
40 int const N = (int)config[5]; // the maximum number of time steps
41 double const gamma = config[6]; // the constant of the perfect gas
42 double const CFL = config[7]; // the CFL number
43 double const h = config[10]; // the length of the initial spatial grids
44 double tau = config[16]; // the length of the time step
45
46 _Bool find_bound = false;
47
48 double Mom, Ene;
49 double c_L, c_R; // the speeds of sound
50 /*
51 * mid: the Riemann solutions.
52 * [rho_star, u_star, p_star]
53 */
54 double dire[3], mid[3];
55
56 double nu; // nu = tau/h
57 double h_S_max; // h/S_max, S_max is the maximum wave speed
58 double time_c = 0.0; // the current time
59 _Bool stop_t = false;
60 int nt = 0; // the number of times storing plotting data
61
62 struct b_f_var bfv_L = {.SRHO = 0.0, .SP = 0.0, .SU = 0.0}, bfv_R = bfv_L; // Left/Right boundary condition
63 struct i_f_var ifv_L = {.gamma = gamma}, ifv_R = ifv_L;
64
65 double ** RHO = CV.RHO;
66 double ** U = CV.U;
67 double ** P = CV.P;
68 double ** E = CV.E;
69 // the numerical flux at (x_{j-1/2}, t_{n}).
70 double * F_rho = (double*)malloc((m+1) * sizeof(double));
71 double * F_u = (double*)malloc((m+1) * sizeof(double));
72 double * F_e = (double*)malloc((m+1) * sizeof(double));
73 if(F_rho == NULL || F_u == NULL || F_e == NULL)
74 {
75 printf("NOT enough memory! Flux\n");
76 goto return_NULL;
77 }
78
79//-----------------------THE MAIN LOOP--------------------------------
80 for(k = 1; k <= N; ++k)
81 {
82 tic = clock();
83 if (time_c >= time_plot[nt] && nt < (*N_plot-1))
84 {
85 for(j = 0; j < m; ++j)
86 {
87 RHO[nt+1][j] = RHO[nt][j];
88 U[nt+1][j] = U[nt][j];
89 E[nt+1][j] = E[nt][j];
90 P[nt+1][j] = P[nt][j];
91 }
92 nt++;
93 }
94
95 h_S_max = INFINITY; // h/S_max = INFINITY
96
97 find_bound = bound_cond_slope_limiter(false, m, nt, &CV, &bfv_L, &bfv_R, find_bound, false, time_c);
98 if(!find_bound)
99 goto return_NULL;
100
101 for(j = 0; j <= m; ++j)
102 { /*
103 * j-1 j j+1
104 * j-1/2 j-1 j+1/2 j j+3/2 j+1
105 * o-----X-----o-----X-----o-----X--...
106 */
107 if(j) // Initialize the initial values.
108 {
109 ifv_L.RHO = RHO[nt][j-1];
110 ifv_L.U = U[nt][j-1];
111 ifv_L.P = P[nt][j-1];
112 }
113 else
114 {
115 ifv_L.RHO = bfv_L.RHO;
116 ifv_L.U = bfv_L.U;
117 ifv_L.P = bfv_L.P;
118 }
119 if(j < m)
120 {
121 ifv_R.RHO = RHO[nt][j];
122 ifv_R.U = U[nt][j];
123 ifv_R.P = P[nt][j];
124 }
125 else
126 {
127 ifv_R.RHO = bfv_R.RHO;
128 ifv_R.U = bfv_R.U;
129 ifv_R.P = bfv_R.P;
130 }
131
132 c_L = sqrt(gamma * ifv_L.P / ifv_L.RHO);
133 c_R = sqrt(gamma * ifv_R.P / ifv_R.RHO);
134 h_S_max = fmin(h_S_max, h/(fabs(ifv_L.U)+fabs(c_L)));
135 h_S_max = fmin(h_S_max, h/(fabs(ifv_R.U)+fabs(c_R)));
136
137//========================Solve Riemann Problem========================
138 linear_GRP_solver_Edir(dire, mid, &ifv_L, &ifv_R, eps, INFINITY);
139
140 if(star_dire_check(mid, dire, 1))
141 {
142 printf(" on [%d, %d] (t_n, x).\n", k, j);
143 stop_t = true;
144 }
145
146 F_rho[j] = mid[0]*mid[1];
147 F_u[j] = F_rho[j]*mid[1] + mid[2];
148 F_e[j] = (gamma/(gamma-1.0))*mid[2] + 0.5*F_rho[j]*mid[1];
149 F_e[j] = F_e[j]*mid[1];
150 }
151
152//====================Time step and grid fixed======================
153 // If no total time, use fixed tau and time step N.
154 if(isfinite(t_all) || !isfinite(config[16]) || config[16] <= 0.0)
155 {
156 tau = CFL * h_S_max;
157 if(tau < eps)
158 {
159 printf("\nThe length of the time step is so small on [%d, %g, %g] (t_n, time_c, tau)\n", k, time_c, tau);
160 stop_t = true;
161 }
162 else if((time_c + tau) > (t_all - eps))
163 tau = t_all - time_c;
164 else if(!isfinite(tau))
165 {
166 printf("NAN or INFinite error on [%d, %g, %g] (t_n, time_c, tau) - CFL\n", k, time_c, tau);
167 goto return_NULL;
168 }
169 }
170 nu = tau / h;
171
172//======================THE CORE ITERATION=========================(On Eulerian Coordinate)
173 for(j = 0; j < m; ++j) // forward Euler
174 { /*
175 * j-1 j j+1
176 * j-1/2 j-1 j+1/2 j j+3/2 j+1
177 * o-----X-----o-----X-----o-----X--...
178 */
179 Mom = RHO[nt][j]*U[nt][j] - nu*(F_u[j+1] -F_u[j]);
180 Ene = RHO[nt][j]*E[nt][j] - nu*(F_e[j+1] -F_e[j]);
181 RHO[nt][j] = RHO[nt][j] - nu*(F_rho[j+1]-F_rho[j]);
182
183 U[nt][j] = Mom / RHO[nt][j];
184 E[nt][j] = Ene / RHO[nt][j];
185 P[nt][j] = (Ene - 0.5*Mom*U[nt][j])*(gamma-1.0);
186 if(P[nt][j] < eps || RHO[nt][j] < eps)
187 {
188 printf("<0.0 error on [%d, %d] (t_n, x) - Update\n", k, j);
189 stop_t = true;
190 }
191 }
192
193//============================Time update=======================
194
195 time_c += tau;
196 if(isfinite(t_all))
197 DispPro(time_c*100.0/t_all, k);
198 else
199 DispPro(k*100.0/N, k);
200 if(stop_t || time_c > (t_all - eps) || !isfinite(time_c))
201 break;
202
203//===========================Fixed variable location=======================
204
205 toc = clock();
206 cpu_time_sum += ((double)toc - (double)tic) / (double)CLOCKS_PER_SEC;
207 cpu_time[nt] = cpu_time_sum;
208 }
209
210 printf("\nTime is up at time step %d.\n", k);
211 printf("The cost of CPU time for 1D-Godunov Eulerian scheme for this problem is %g seconds.\n", cpu_time_sum);
212//---------------------END OF THE MAIN LOOP----------------------
213
214return_NULL:
215 config[5] = (double)k;
216 *N_plot = nt+1;
217 if(isfinite(time_c))
218 time_plot[nt] = time_c;
219 else if(isfinite(t_all))
220 time_plot[nt] = t_all;
221 else if(isfinite(tau))
222 time_plot[nt] = k*tau;
223
224 free(F_rho);
225 free(F_u);
226 free(F_e);
227 F_rho = NULL;
228 F_u = NULL;
229 F_e = NULL;
230}
