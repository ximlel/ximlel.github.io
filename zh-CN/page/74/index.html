137 * o-----X-----o-----X-----o-----X--...
138 */

140 {

142 ifv_L.U = U[nt][j-1] + 0.5*h*s_u[j-1];
143 ifv_L.P = P[nt][j-1] + 0.5*h*s_p[j-1];
144 }
145 else
146 {
147 ifv_L.RHO = bfv_L.RHO + 0.5*h*bfv_L.SRHO;
148 ifv_L.U = bfv_L.U + 0.5*h*bfv_L.SU;
149 ifv_L.P = bfv_L.P + 0.5*h*bfv_L.SP;
150 }
151 if(j < m)
152 {
153 ifv_R.RHO = RHO[nt][j] - 0.5*h*s_rho[j];
154 ifv_R.U = U[nt][j] - 0.5*h*s_u[j];
155 ifv_R.P = P[nt][j] - 0.5*h*s_p[j];
156 }
157 else
158 {
159 ifv_R.RHO = bfv_R.RHO + 0.5*h*bfv_R.SRHO;
160 ifv_R.U = bfv_R.U + 0.5*h*bfv_R.SU;
161 ifv_R.P = bfv_R.P + 0.5*h*bfv_R.SP;
162 }
163
164 c_L = sqrt(gamma * ifv_L.P / ifv_L.RHO);
165 c_R = sqrt(gamma * ifv_R.P / ifv_R.RHO);
166 h_S_max = fmin(h_S_max, h/(fabs(ifv_L.U)+fabs(c_L)));
167 h_S_max = fmin(h_S_max, h/(fabs(ifv_R.U)+fabs(c_R)));
168

171 ifv_L.d_u = s_u[j-1];
172 ifv_L.d_p = s_p[j-1];
173 ifv_L.d_rho = s_rho[j-1];
174 }
175 else

178 ifv_L.d_u = bfv_L.SU;
179 ifv_L.d_p = bfv_L.SP;

181 if(j < m)

184 ifv_R.d_p = s_p[j];
185 ifv_R.d_rho = s_rho[j];
186 }

190 ifv_R.d_u = bfv_R.SU;
191 ifv_R.d_p = bfv_R.SP;
192 }

221 if(tau < eps)

239 U_next[j] += 0.5 * tau * U_t[j];
240 P_next[j] += 0.5 * tau * P_t[j];
241
242 F_rho[j] = RHO_next[j]*U_next[j];
243 F_u[j] = F_rho[j]*U_next[j] + P_next[j];
244 F_e[j] = (gamma/(gamma-1.0))*P_next[j] + 0.5*F_rho[j]*U_next[j];
245 F_e[j] = F_e[j]*U_next[j];
246
247 RHO_next[j] += 0.5 * tau * RHO_t[j];
248 U_next[j] += 0.5 * tau * U_t[j];
249 P_next[j] += 0.5 * tau * P_t[j];
250 }
251
252//======================THE CORE ITERATION=========================(On Eulerian Coordinate)
253 for(j = 0; j < m; ++j) // forward Euler
254 { /*
255 * j-1 j j+1
256 * j-1/2 j-1 j+1/2 j j+3/2 j+1
257 * o-----X-----o-----X-----o-----X--...
258 */
259 Mom = RHO[nt][j]*U[nt][j] - nu*(F_u[j+1] -F_u[j]);
260 Ene = RHO[nt][j]*E[nt][j] - nu*(F_e[j+1] -F_e[j]);
261 RHO[nt][j] = RHO[nt][j] - nu*(F_rho[j+1]-F_rho[j]);
262
263 U[nt][j] = Mom / RHO[nt][j];
264 E[nt][j] = Ene / RHO[nt][j];
265 P[nt][j] = (Ene - 0.5*Mom*U[nt][j])*(gamma-1.0);
266
267 if(P[nt][j] < eps || RHO[nt][j] < eps)
268 {
269 printf("<0.0 error on [%d, %d] (t_n, x) - Update\n", k, j);
270 stop_t = true;
271 }
272
273//============================compute the slopes============================
274 s_u[j] = ( U_next[j+1] - U_next[j])/h;
275 s_p[j] = ( P_next[j+1] - P_next[j])/h;
276 s_rho[j] = (RHO_next[j+1] - RHO_next[j])/h;
277 }
278
279//============================Time update=======================
280
281 time_c += tau;
282 if(isfinite(t_all))
283 DispPro(time_c*100.0/t_all, k);
284 else
285 DispPro(k*100.0/N, k);
286 if(stop_t || time_c > (t_all - eps) || !isfinite(time_c))

292 cpu_time_sum += ((double)toc - (double)tic) / (double)CLOCKS_PER_SEC;

294 }

296 printf("\nTime is up at time step %d.\n", k);
297 printf("The cost of CPU time for 1D-GRP Eulerian scheme for this problem is %g seconds.\n", cpu_time_sum);

[in]schemeScheme name.
  • argv[1]: Folder name of test example (input path).
  • argv[3]: Order of numerical scheme[_scheme name] (= 1[_Riemann_exact] or 2[_GRP]).
  • argv[argc_least+1,argc_least+2,…]: Configuration supplement config[n]=(double)C (= n=C).
