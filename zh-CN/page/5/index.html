This function use GRP scheme to solve radially(cylindrically) symmetric compressible flows of motion on Lagrangian coordinate.

参数
[in,out]CVStructure of cell variable data.
[in,out]rmvStructure of radially symmetric meshing variable data.
[in]RArray of the r-coordinate data.
[in]MSpatial dimension number for radially symmetric flow.
  • M=1: planar flow.
  • M=2: cylindrical flow. (√)
  • M=3: spherical flow.
[out]cpu_timeArray of the CPU time recording.
[in]problemName of the numerical results for the test problem.
[in]N_TNumber of data dimension storing fluid variables in memory.
[in,out]N_plotPointer to the number of time steps for plotting.
[in,out]time_plotArray of the plotting time recording.
参见
Theory is found in Reference [1].
[1] R. Chen, J. Li & B. Tian, Application of the GRP Scheme for Cylindrical Compressible Fluid Flows. COMMUNICATIONS IN COMPUTATIONAL PHYSICS, 24.5: 1523-1555, 2018.

260 Umin[i+1] += 0.5 * dt * U_t[i+1];
261 Pmin[i+1] += 0.5 * dt * P_t[i+1];
262
263 F_u[i+1] = Pmin[i+1];
264 F_e[i+1] = Pmin[i+1] * Umin[i+1];
265
266 Rb_NStep = Rb[i+1] + Umin[i+1]*dt;
267 Lb_NStep = 2.0 * Rb_NStep * tan(0.5*dtheta);
268 Lbh[i+1] = 0.5 * (Lb[i+1] + Lb_NStep);
269 Rbh[i+1] = (Rb[i+1]*(2.*Lb[i+1]+Lb_NStep)+Rb_NStep*(Lb[i+1]+2.*Lb_NStep))/(3.*(Lb[i+1]+Lb_NStep));
270 Rb[i+1] = Rb_NStep;
271 Lb[i+1] = Lb_NStep;
272 RR[i] = Rb[i+1]-(2.*Lb[i]+Lb[i+1])/(3.*(Lb[i]+Lb[i+1]))*(Rb[i+1]-Rb[i]);
273 /*
274 Rb_side[i] =0.25*(Lb[i]+Lb[i+1]) /sin(0.5*dtheta);
275 Lb_side[i] =0.5 *(Lb[i+1]-Lb[i]) /sin(0.5*dtheta);
276 Rbh_side[i]=0.25*(Lbh[i]+Lbh[i+1])/sin(0.5*dtheta);
277 Lbh_side[i]=0.5 *(Lbh[i+1]-Lbh[i])/sin(0.5*dtheta);
278 Sh[i]=Rbh[i+1]*Lbh[i+1]-Rbh[i]*Lbh[i]-sin(dtheta)*Rbh_side[i]*Lbh_side[i];
279 */
280
281 Umin[i+1] += dt * U_t[i+1] * 0.5;
282 Pmin[i+1] += dt * P_t[i+1] * 0.5;
283 DLmin[i+1] += dt * DL_t[i+1];
284 DRmin[i+1] += dt * DR_t[i+1];
285 }

287 radial_mesh_update(rmv);
288
289 DD[0] = mass[0]/vol[0];
290 UU[0] = 0.0;
291 EE[0] = EE[0] - dt/mass[0]*(F_e[1]*Rbh[1]*Lbh[1]);
292 for(i = 1; i <= Ncell; i++) //m=2
293 {
294 DD[i] = mass[i]/vol[i];
295 UU[i] = UU[i] - dt/mass[i]*((F_u[i+1]-F_u2[i])*Rbh[i+1]*Lbh[i+1]-(F_u[i]-F_u2[i])*Rbh[i]*Lbh[i]);
296 EE[i] = EE[i] - dt/mass[i]*( F_e[i+1] *Rbh[i+1]*Lbh[i+1]- F_e[i] *Rbh[i]*Lbh[i]);
297 }
298 for(i = 0; i <= Ncell; i++) //m=2
299 {
300 PP[i] = (EE[i] - 0.5*UU[i]*UU[i]) * (GammaGamma[i]-1.0) * DD[i];
301 if (PP[i] < eps)
302 {
303 printf("p<0.0 error on [%d, %d] (t_n, x) - Update\n", k, i);
304 stop_t = true;
305 }
306 }
307
308 for(i = 1; i <= Ncell; i++)
309 {
310 DmU[i]=(Umin[i+1] -Umin[i]) /Ddr[i];
311 DmP[i]=(Pmin[i+1] -Pmin[i]) /Ddr[i];
312 DmD[i]=(DLmin[i+1]-DRmin[i])/Ddr[i];
313 }
314 DmU[0] =(Umin[1]-UU[0]) /dRc[0];
315 DmP[0] =(Pmin[1]-PP[0]) /dRc[0];
316 DmD[0] =(DLmin[1]-DD[0])/dRc[0];
317 DmU[Ncell+1]=(Umin[Ncell+1] -UU[Ncell])/dRc[Ncell];
318 DmP[Ncell+1]=(Pmin[Ncell+1] -PP[Ncell])/dRc[Ncell];
319 DmD[Ncell+1]=(DLmin[Ncell+1]-DD[Ncell])/dRc[Ncell];
320
321 time_c=time_c+dt;
322 if(isfinite(Timeout))
323 DispPro(time_c*100.0/Timeout, k);
324 else
325 DispPro(k*100.0/N, k);
326 if(stop_t || time_c > (Timeout - eps) || !isfinite(time_c))
327 break;
328

330 cpu_time_sum += ((double)toc - (double)tic) / (double)CLOCKS_PER_SEC;
331 cpu_time[nt] = cpu_time_sum;
332 }

334 printf("\nTime is up at time step %d.\n", k);
335 printf("The cost of CPU time for 1D-GRP Eulerian scheme for this problem is %g seconds.\n", cpu_time_sum);
