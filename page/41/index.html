A Quasi-1D direct Eulerian GRP solver for unsteady compressible inviscid two-component flow in two space dimension.

参数
[out]wave_speedthe velocity of left and right waves.
[out]Dthe temporal derivative of fluid variables.
[rho, u, v, p, phi, z_a]_t
[out]Uthe intermediate Riemann solutions at t-axis.
[rho_mid, u_mid, v_mid, p_mid, phi_mid, z_a_mid]
[out]U_starthe Riemann solutions in star region.
[rho_star_L, u_star, rho_star_R, p_star, c_star_L, c_star_R]
[in]ifv_LLeft States (rho/u/v/p/phi/z, d_, t_, gammaL).
[in]ifv_RRight States (rho/u/v/p/phi/z, d_, t_, gammaR).
  • s_: normal derivatives.
  • t_: tangential derivatives.
  • gamma: the constant of the perfect gas.
[in]epsthe largest value could be seen as zero.
[in]atcParameter that determines the solver type.
  • INFINITY: acoustic approximation
    • ifv_.s_, ifv_.t_ = -0.0: exact Riemann solver
  • eps: Quasi-1D GRP solver(nonlinear + acoustic case)
    • ifv_.t_ = -0.0: Planar-1D GRP solver
  • -0.0: Quasi-1D GRP solver(only nonlinear case)
    • ifv_.t_ = -0.0: Planar-1D GRP solver
参见
Theory is found in Reference [1].
[1] M. Ben-Artzi, J. Li & G. Warnecke, A direct Eulerian GRP scheme for compressible fluid flows. Journal of Computational Physics, 218.1: 19-43, 2006.

460 L_u = -rho_R * (H2*c_R*c_R + H3) * d_u_R;
461 L_p = H2 * SmUR * d_p_R;
462

464 D[0] = (D[0] - u_star*(L_p+L_rho+L_u)) / SmUs;
465
466 f = SmUR*(H2*d_p_R + H3*d_rho_R) - rho_R*(H2*c_R*c_R+H3)*d_u_R;
467 rho_x = (f + H1*(p_t_mat - rho_star_R*SmUs*u_t_mat) - D[0]) / (SmUR+u_R);//shk_spd;
468 D[0] = D[0] + lambda_u*rho_x;
469
470 D[2] = -U[1] * SmUR * d_v_R / SmUs;
471 D[2] = D[2] + lambda_u*d_v_R;
472 D[4] = -U[1] * SmUR * d_z_R / SmUs;
473 D[4] = D[4] + lambda_u*d_z_R;
474 D[5] = -U[1] * SmUR * d_phi_R / SmUs;
475 D[5] = D[5] + lambda_u*d_phi_R;
476 }
477 }
478 else //the direction is between the 1-wave and the contact discontinuety
479 {
480 if(CRW[0]) //the 1-wave is a CRW
481 {
482 //already total D!
483 D[0] = rho_star_L*(u_star-lambda_u)*pow(c_star_L/c_L, (1.0+zetaL)/zetaL)*(d_p_L - d_rho_L*c_L*c_L)/rho_L;
484 D[0] = (D[0] + D[3]) / c_star_L/c_star_L;
485
486 D[2] = -U[1]*d_v_L*U[0]/rho_L;
487 D[2] = D[2] + lambda_u*d_v_L;
488 D[4] = -U[1]*d_z_L*U[0]/rho_L;
489 D[4] = D[4] + lambda_u*d_z_L;
490 D[5] = -U[1]*d_phi_L*U[0]/rho_L;
491 D[5] = D[5] + lambda_u*d_phi_L;
492 }
493 else //the 1-wave is a shock
494 {
495 SmUs = -sqrt(0.5*((gammaL+1.0)*p_L +(gammaL-1.0)*p_star)/rho_star_L);
496 SmUL = -sqrt(0.5*((gammaL+1.0)*p_star+(gammaL-1.0)*p_L )/rho_L);
497
498 VAR = p_L + zetaL*p_star;
499 H1 = rho_L * p_L * (1.0 - zetaL*zetaL) / VAR/VAR;
500 H2 = rho_L * p_star * (zetaL*zetaL - 1.0) / VAR/VAR;
501 H3 = (p_star + zetaL*p_L)/VAR;
502
503 L_rho = SmUL * H3 * d_rho_L;
504 L_u = -rho_L*(H2*c_L*c_L + H3) * d_u_L;
505 L_p = H2 * SmUL * d_p_L;
506
507 D[0] = ((u_star+SmUs)/c_star_L/c_star_L - H1*u_star)*p_t_mat + rho_star_L*u_star*SmUs*H1*u_t_mat;
508 D[0] = (D[0] - u_star*(L_p+L_rho+L_u))/ SmUs;
509
510 f = SmUL*(H2*d_p_L + H3*d_rho_L) - rho_L*(H2*c_L*c_L+H3)*d_u_L;
511 rho_x = (f + H1*(p_t_mat - rho_star_L*SmUs*u_t_mat) - D[0]) / (SmUL+u_L);
512 D[0] = D[0] + lambda_u*rho_x;
513
514 D[2] = -U[1] * SmUL * d_v_L / SmUs;
515 D[2] = D[2] + lambda_u*d_v_L;
516 D[4] = -U[1] * SmUL * d_z_L / SmUs;
517 D[4] = D[4] + lambda_u*d_z_L;
518 D[5] = -U[1] * SmUL * d_phi_L / SmUs;
519 D[5] = D[5] + lambda_u*d_phi_L;
520 }
521 }
522 //--end of non-sonic case--
523 }
524 T_p = 0.5*((t_u_L*(U[0]*C) + t_p_L) - (t_u_R*(U[0]*C) - t_p_R));
525 T_u = 0.5*(t_u_L + t_p_L/(U[0]*C) + t_u_R - t_p_R/(U[0]*C));
526 if (u_star > lambda_u)
527 {
528 T_rho = t_rho_L - t_p_L/(C*C) + T_p/(C*C);
529 D[0] = D[0] - (U[2]-lambda_v)*T_rho - U[0]*t_v_L;
530 D[1] = D[1] - (U[2]-lambda_v)*T_u;
531 D[2] = D[2] - (U[2]-lambda_v)*t_v_L - T_p/U[0];
532 D[3] = D[3] - (U[2]-lambda_v)*T_p - U[0]*C*C*t_v_L;
533 D[4] = D[4] - (U[2]-lambda_v)*t_z_L;
534 D[5] = D[5] - (U[2]-lambda_v)*t_phi_L;
535 }
536 else
537 {
538 T_rho = t_rho_R - t_p_R/(C*C) + T_p/(C*C);
539 D[0] = D[0] - (U[2]-lambda_v)*T_rho - U[0]*t_v_R;
540 D[1] = D[1] - (U[2]-lambda_v)*T_u;
541 D[2] = D[2] - (U[2]-lambda_v)*t_v_R - T_p/U[0];
542 D[3] = D[3] - (U[2]-lambda_v)*T_p - U[0]*C*C*t_v_R;
543 D[4] = D[4] - (U[2]-lambda_v)*t_z_R;
544 D[5] = D[5] - (U[2]-lambda_v)*t_phi_R;
545 }
546 //----end of non-trivial case----
547 }
548 U_star[0] = rho_star_L;
549 U_star[1] = u_star;
550 U_star[2] = rho_star_R;
551 U_star[3] = p_star;
552 U_star[4] = c_star_L;
553 U_star[5] = c_star_R;
554}
