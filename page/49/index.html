参数

34 C_R = sqrt(gamma*P_R/RHO_R);
35 double z = 0.5 * (gamma-1.0) / gamma;

37 P_star = pow((C_L + C_R - (gamma-1.0)/2.0*(U_R - U_L))/(C_L/pow(P_L,z) + C_R/pow(P_R,z)), 1.0/z);

39 U_star_L = U_L + 2.0/(gamma - 1.0)*(C_L - C_star_L);

44 lambda_L_3 = U_star_R + C_star_R;
45 lambda_R_3 = U_R + C_R;

48 H_L = gamma/(gamma-1.0)*P_L/RHO_L + 0.5*(U_L*U_L);

53 F[2] = 0.5*(RHO_L*U_L*H_L+RHO_R*U_R*H_R);

58 H_S = (H_L*sqrt(RHO_L)+H_R*sqrt(RHO_R)) / (sqrt(RHO_L)+sqrt(RHO_R));
59 C_S = sqrt((gamma-1.0)*(H_S-0.5*U_S*U_S));

75void GRP_solver_2D_EUL_source(const int m, const int n, struct cell_var_stru * CV, double ** X, double **Y,
76 double * cpu_time, const char * problem, int N_T, int * N_plot, double time_plot[])
77{
78#ifdef _OPENMP
79 printf("@@ Number of threads for OpenMP: %d\n", omp_get_max_threads());
80#endif
81#ifdef _OPENACC
82 printf("@@ Number of CPU devices for OpenACC: %d\n", acc_get_num_devices(acc_device_host));
83 printf("@@ Number of GPU devices for OpenACC: %d\n", acc_get_num_devices(acc_device_not_host));
84#endif
85 /*
86 * i is a frequently used index for y-spatial variables.
87 * j is a frequently used index for x-spatial variables.
88 * k is a frequently used index for the time step.
89 */
90 int i, j, k = 0;
91
92 double tic, toc;
93 double cpu_time_sum = 0.0;
94
95 double const t_all = config[1]; // the total time
96 double const eps = config[4]; // the largest value could be seen as zero
97 int const N = (int)config[5]; // the maximum number of time steps
98 double const gamma = config[6]; // the constant of the perfect gas
99 double const CFL = config[7]; // the CFL number
100 double const h_x = config[10]; // the length of the initial x-spatial grids
101 double const h_y = config[11]; // the length of the initial y-spatial grids
102 double tau = config[16]; // the length of the time step
103
104 _Bool find_bound_x = false, find_bound_y = false;
105 int flux_err;
106
107 double mom_x, mom_y, ene;
108 double c; // the speeds of sound
109
110 double mu, nu; // nu = tau/h_x, mu = tau/h_y.
111 double h_S_max, sigma; // h/S_max, S_max is the maximum character speed, sigma is the character speed
112 double time_c = 0.0; // the current time
113 _Bool stop_t = false;
114 int nt = 0, nt_plot = 0; // the number of times storing plotting data
115
116 // Left/Right/Upper/Downside boundary condition
117 struct b_f_var * bfv_L = NULL, * bfv_R = NULL, * bfv_U = NULL, * bfv_D = NULL;
118 // the slopes of variable values.
119 INIT_MEM_2D(s_rho, m, n); INIT_MEM_2D(t_rho, m, n);
120 INIT_MEM_2D(s_u, m, n); INIT_MEM_2D(t_u, m, n);
121 INIT_MEM_2D(s_v, m, n); INIT_MEM_2D(t_v, m, n);
122 INIT_MEM_2D(s_p, m, n); INIT_MEM_2D(t_p, m, n);
123 // the variable values at (x_{j-1/2}, t_{n+1}).
124 INIT_MEM_2D(rhoIx, m+1, n);
125 INIT_MEM_2D(uIx, m+1, n);
126 INIT_MEM_2D(vIx, m+1, n);
127 INIT_MEM_2D(pIx, m+1, n);
128 INIT_MEM_2D(F_rho, m+1, n);
129 INIT_MEM_2D(F_u, m+1, n);
130 INIT_MEM_2D(F_v, m+1, n);
131 INIT_MEM_2D(F_e, m+1, n);
132 // the variable values at (y_{j-1/2}, t_{n+1}).
133 INIT_MEM_2D(rhoIy, m, n+1);
134 INIT_MEM_2D(uIy, m, n+1);
135 INIT_MEM_2D(vIy, m, n+1);
136 INIT_MEM_2D(pIy, m, n+1);
137 INIT_MEM_2D(G_rho, m, n+1);
138 INIT_MEM_2D(G_u, m, n+1);
139 INIT_MEM_2D(G_v, m, n+1);
140 INIT_MEM_2D(G_e, m, n+1);
141 // boundary condition
142 BC_INIT_MEM_1D(bfv_L, n); BC_INIT_MEM_1D(bfv_R, n);
143 BC_INIT_MEM_1D(bfv_D, m); BC_INIT_MEM_1D(bfv_U, m);
144
145//------------THE MAIN LOOP-------------
146 for(k = 1; k <= N; ++k)
147 {
148#ifdef _OPENMP
149 tic = omp_get_wtime();
150#elif defined _OPENACC
151 tic = omp_get_wtime();
152#else
153 tic = (double)clock() / (double)CLOCKS_PER_SEC;
154#endif
155 if (time_c >= time_plot[nt_plot] && nt_plot < (*N_plot-1))
156 {
157#ifndef NOTECPLOT
158 file_2D_write_POINT_TEC(m, n, 1, CV + nt, X, Y, cpu_time, problem, time_plot + nt_plot);
159#endif
160 nt_plot++;
161 if (nt < (N_T-1))
162 {
163 for(j = 0; j < m; ++j)
164 for(i = 0; i < n; ++i)
165 {
166 CV[nt+1].RHO[j][i] = CV[nt].RHO[j][i];
167 CV[nt+1].U[j][i] = CV[nt].U[j][i];
168 CV[nt+1].V[j][i] = CV[nt].V[j][i];
169 CV[nt+1].E[j][i] = CV[nt].E[j][i];
170 CV[nt+1].P[j][i] = CV[nt].P[j][i];
171 }
172 nt++;
173 }
174 }
175
176 /* evaluate f and a at some grid points for the iteration
177 * and evaluate the character speed to decide the length
178 * of the time step by (tau * speed_max)/h = CFL
179 */
180 h_S_max = INFINITY; // h/S_max = INFINITY
181
182 for(j = 0; j < m; ++j)
183 for(i = 0; i < n; ++i)
184 {
185 c = sqrt(gamma * CV->P[j][i] / CV->RHO[j][i]);
186 sigma = fabs(c) + fabs(CV->U[j][i]) + fabs(CV->V[j][i]);
187 h_S_max = fmin(h_S_max, fmin(h_x,h_y) / sigma);
188 }
189 // If no total time, use fixed tau and time step N.
190 if(isfinite(t_all) || !isfinite(config[16]) || config[16] <= 0.0)
191 {
192 tau = CFL * h_S_max;
193 if(tau < eps)
194 {
195 printf("\nThe length of the time step is so small on [%d, %g, %g] (t_n, time_c, tau)\n", k, time_c, tau);
196 stop_t = true;
197 }
198 else if((time_c + tau) > (t_all - eps))
199 tau = t_all - time_c;
200 else if(!isfinite(tau))
201 {
202 printf("NAN or INFinite error on [%d, %g, %g] (t_n, time_c, tau) - CFL\n", k, time_c, tau);
203 goto return_NULL;
204 }
205 }
206 nu = tau / h_x;
207 mu = tau / h_y;
208
209 find_bound_x = bound_cond_slope_limiter_x(m, n, nt, CV, bfv_L, bfv_R, bfv_D, bfv_U, find_bound_x, true, time_c);
210 if(!find_bound_x)
211 goto return_NULL;
212 find_bound_y = bound_cond_slope_limiter_y(m, n, nt, CV, bfv_L, bfv_R, bfv_D, bfv_U, find_bound_y, true, time_c);
213 if(!find_bound_y)
214 goto return_NULL;
215
216 flux_err = flux_generator_x(m, n, nt, tau, CV, bfv_L, bfv_R, true);
217 if(flux_err == 1)
218 goto return_NULL;
219 else if(flux_err == 2)
220 stop_t = true;
221 flux_err = flux_generator_y(m, n, nt, tau, CV, bfv_D, bfv_U, true);
222 if(flux_err == 1)
223 goto return_NULL;
224 else if(flux_err == 2)
225 stop_t = true;
226
227//===============THE CORE ITERATION=================
228#ifdef _OPENMP
229#pragma omp parallel for private(mom_x, mom_y, ene) collapse(2)
230#elif defined _OPENACC
231#pragma acc parallel loop private(mom_x, mom_y, ene) collapse(2)
232#endif
233 for(i = 0; i < n; ++i)
234 for(j = 0; j < m; ++j)
235 { /*
236 * j-1 j j+1
237 * j-1/2 j-1 j+1/2 j j+3/2 j+1
238 * o-----X-----o-----X-----o-----X--...
239 */
240 mom_x = CV[nt].RHO[j][i]*CV[nt].U[j][i] - nu*(CV->F_u[j+1][i] -CV->F_u[j][i]) - mu*(CV->G_u[j][i+1] -CV->G_u[j][i]);
241 mom_y = CV[nt].RHO[j][i]*CV[nt].V[j][i] - nu*(CV->F_v[j+1][i] -CV->F_v[j][i]) - mu*(CV->G_v[j][i+1] -CV->G_v[j][i]);
242 ene = CV[nt].RHO[j][i]*CV[nt].E[j][i] - nu*(CV->F_e[j+1][i] -CV->F_e[j][i]) - mu*(CV->G_e[j][i+1] -CV->G_e[j][i]);
243 CV[nt].RHO[j][i] += - nu*(CV->F_rho[j+1][i]-CV->F_rho[j][i]) - mu*(CV->G_rho[j][i+1]-CV->G_rho[j][i]);
244
245 CV[nt].U[j][i] = mom_x / CV[nt].RHO[j][i];
246 CV[nt].V[j][i] = mom_y / CV[nt].RHO[j][i];
247 CV[nt].E[j][i] = ene / CV[nt].RHO[j][i];
248 CV[nt].P[j][i] = (ene - 0.5*mom_x*CV[nt].U[j][i] - 0.5*mom_y*CV[nt].V[j][i])*(gamma-1.0);
249 if(CV[nt].P[j][i] < eps || CV[nt].RHO[j][i] < eps)
250 {
251 printf("<0.0 error on [%d, %d, %d] (t_n, x, y) - Update\n", k, j, i);
252 stop_t = true;
253 }
254
255 CV->s_rho[j][i] = (CV->rhoIx[j+1][i] - CV->rhoIx[j][i])/h_x;
256 CV->s_u[j][i] = ( CV->uIx[j+1][i] - CV->uIx[j][i])/h_x;
257 CV->s_v[j][i] = ( CV->vIx[j+1][i] - CV->vIx[j][i])/h_x;
258 CV->s_p[j][i] = ( CV->pIx[j+1][i] - CV->pIx[j][i])/h_x;
259 CV->t_rho[j][i] = (CV->rhoIy[j][i+1] - CV->rhoIy[j][i])/h_y;
260 CV->t_u[j][i] = ( CV->uIy[j][i+1] - CV->uIy[j][i])/h_y;
261 CV->t_v[j][i] = ( CV->vIy[j][i+1] - CV->vIy[j][i])/h_y;
262 CV->t_p[j][i] = ( CV->pIy[j][i+1] - CV->pIy[j][i])/h_y;
263 } // End of parallel region
264
265//==================================================
266
267 time_c += tau;
268 if(isfinite(t_all))
269 DispPro(time_c*100.0/t_all, k);
270 else
271 DispPro(k*100.0/N, k);
272 if(stop_t || time_c > (t_all - eps) || !isfinite(time_c))
273 break;
274
275 //===========================Fixed variable location=======================
276
277#ifdef _OPENMP
278 toc = omp_get_wtime();
279#elif defined _OPENACC
280 toc = omp_get_wtime();
281#else
282 toc = (double)clock() / (double)CLOCKS_PER_SEC;
283#endif
284 cpu_time_sum += toc - tic;
285 cpu_time[nt] = cpu_time_sum;
286 }
287
288 printf("\nTime is up at time step %d.\n", k);
289 printf("The cost of CPU time for genuinely 2D-GRP Eulerian scheme without dimension splitting for this problem is %g seconds.\n", cpu_time_sum);
290 //------------END OF THE MAIN LOOP-------------
291
292return_NULL:
293 config[5] = (double)k;
294 *N_plot = nt+1;
295 if(isfinite(time_c))
296 time_plot[nt] = time_c;
297 else if(isfinite(t_all))
298 time_plot[nt] = t_all;
299 else if(isfinite(tau))
300 time_plot[nt] = k*tau;
301
302 for(j = 0; j < m+1; ++j)
303 {
304 free(CV->F_rho[j]); free(CV->F_u[j]); free(CV->F_v[j]); free(CV->F_e[j]);
305 free(CV->rhoIx[j]); free(CV->uIx[j]); free(CV->vIx[j]); free(CV->pIx[j]);
306 CV->F_rho[j]= NULL; CV->F_u[j]= NULL; CV->F_v[j]= NULL; CV->F_e[j]= NULL;
307 CV->rhoIx[j]= NULL; CV->uIx[j]= NULL; CV->vIx[j]= NULL; CV->pIx[j]= NULL;
308 }
309 for(j = 0; j < m; ++j)
310 {
311 free(CV->G_rho[j]); free(CV->G_u[j]); free(CV->G_v[j]); free(CV->G_e[j]);
312 free(CV->rhoIy[j]); free(CV->uIy[j]); free(CV->vIy[j]); free(CV->pIy[j]);
313 free(CV->s_rho[j]); free(CV->s_u[j]); free(CV->s_v[j]); free(CV->s_p[j]);
314 free(CV->t_rho[j]); free(CV->t_u[j]); free(CV->t_v[j]); free(CV->t_p[j]);
315
316 CV->G_rho[j]= NULL; CV->G_u[j]= NULL; CV->G_v[j]= NULL; CV->G_e[j]= NULL;
317 CV->rhoIy[j]= NULL; CV->uIy[j]= NULL; CV->vIy[j]= NULL; CV->pIy[j]= NULL;
318 CV->s_rho[j]= NULL; CV->s_u[j]= NULL; CV->s_v[j]= NULL; CV->s_p[j]= NULL;
319 CV->t_rho[j]= NULL; CV->t_u[j]= NULL; CV->t_v[j]= NULL; CV->t_p[j]= NULL;
320 }
321 free(CV->F_rho); free(CV->F_u); free(CV->F_v); free(CV->F_e);
322 free(CV->rhoIx); free(CV->uIx); free(CV->vIx); free(CV->pIx);
323 free(CV->G_rho); free(CV->G_u); free(CV->G_v); free(CV->G_e);
324 free(CV->rhoIy); free(CV->uIy); free(CV->vIy); free(CV->pIy);
325 free(CV->s_rho); free(CV->s_u); free(CV->s_v); free(CV->s_p);
326 free(CV->t_rho); free(CV->t_u); free(CV->t_v); free(CV->t_p);
327 free(bfv_L); free(bfv_R);
328 free(bfv_D); free(bfv_U);
329
330 CV->F_rho= NULL; CV->F_u= NULL; CV->F_v= NULL; CV->F_e= NULL;
331 CV->rhoIx= NULL; CV->uIx= NULL; CV->vIx= NULL; CV->pIx= NULL;
332 CV->G_rho= NULL; CV->G_u= NULL; CV->G_v= NULL; CV->G_e= NULL;
333 CV->rhoIy= NULL; CV->uIy= NULL; CV->vIy= NULL; CV->pIy= NULL;
334 CV->s_rho= NULL; CV->s_u= NULL; CV->s_v= NULL; CV->s_p= NULL;
335 CV->t_rho= NULL; CV->t_u= NULL; CV->t_v= NULL; CV->t_p= NULL;
336 bfv_L= NULL; bfv_R= NULL;
337 bfv_D= NULL; bfv_U= NULL;
338}
