The purpose of this function is to solve the Riemann problem exactly, for the time dependent one dimensional Euler equations for two-component γ-law gas.

参数

[in]u_R,p_R,c_RInitial Velocity/Pressure/sound_speed on right state.
[in]gammaL,gammaRRatio of specific heats.
[out]CRWCentred Rarefaction Wave (CRW) Indicator of left and right waves.
  • true: CRW
  • false: Shock wave
[in]epsThe largest value can be seen as zero.
[in]tolCondition value of 'gap' at the end of the iteration.
[in]NMaximum iteration step.
返回

45 muR = (gammaR-1.0) / (2.0*gammaR);
46 nuR = (gammaR+1.0) / (2.0*gammaR);
47
48 //=====find out the kinds of the 1-wave and the 3-wave, page 132 in the GRP book
49 //find out where (u_LR,p_R) lies on the curve of LEFT state
50 if(p_R > p_L) // (u_LR,p_R) lies on the shock branch of I1
51 {
52 delta_p = p_R - p_L;
53 u_LR = sqrt(1.0 + nuL*delta_p/p_L);
54 u_LR = delta_p * c_L / gammaL / p_L / u_LR;
55 u_LR = u_L - u_LR;
56 }
57 else // (u_LR,p_R) lies on the rarefaction branch of I1
58 {
59 u_LR = pow(p_R/p_L, muL) - 1.0;
60 u_LR = 2.0 * c_L * u_LR / (gammaL-1.0);
61 u_LR = u_L - u_LR;
62 }
63 //find out where (u_RL,p_L) lies on the curve of RIGHT state
64 if(p_L > p_R) // (u_RL, p_L) lies on the shock branch of I3
65 {
66 delta_p = p_L - p_R;
67 u_RL = sqrt(1.0 + nuR*delta_p/p_R);
68 u_RL = delta_p * c_R / gammaR / p_R / u_RL;
69 u_RL = u_R + u_RL;
70 }
71 else // (u_RL, p_L) lies on the rarefaction branch of I3
72 {
73 u_RL = pow(p_L/p_R, muR) - 1.0;
74 u_RL = 2.0 * c_R * u_RL / (gammaR-1.0);
75 u_RL = u_R + u_RL;
76 }
77 if(u_LR > u_R+eps)
78 CRW[1] = false;
79 else
80 CRW[1] = true;
81 if(u_RL > u_L-eps)
82 CRW[0] = true;
83 else
84 CRW[0] = false;
85
86 //======one step of the Newton ietration to get the intersection point of I1 and I3====
87 k1 = -c_L / p_L / gammaL;//the (p,u)-tangent slope on I1 at (u_L,p_L), i.e. [du/dp](p_L)
88 k3 = c_R / p_R / gammaR;//the (p,u)-tangent slope on I3 at (u_R,p_R), i.e. [du/dp](p_R)
89 //the intersect of (u-u_L)=k1*(p-p_L) and (u-u_R)=k3*(p-p_R)
90 p_INT = (k1*p_L - k3*p_R - u_L + u_R) / (k1 - k3);
91 if(p_INT < 0)
92 p_INT = (p_L<p_R)? p_L : p_R;
93 p_INT = 0.5*p_INT;
94
95 //=======compute the gap between U^n_R and U^n_L(see Appendix C)=======
96 if(p_INT > p_L)
97 {
98 delta_p = p_INT - p_L;
99 v_L = sqrt(1.0 + nuL*delta_p/p_L);
100 v_L = delta_p * c_L / gammaL / p_L / v_L;
101 v_L = u_L - v_L;
102 }
103 else
104 {
105 v_L = pow(p_INT/p_L, muL) - 1.0;
106 v_L = 2.0 * c_L * v_L / (gammaL-1.0);
107 v_L = u_L - v_L;
108 }
109 if(p_INT > p_R)
110 {
111 delta_p = p_INT - p_R;
112 v_R = sqrt(1.0 + nuR*delta_p/p_R);
113 v_R = delta_p * c_R / gammaR / p_R / v_R;
114 v_R = u_R + v_R;
115 }
116 else
117 {
118 v_R = pow(p_INT/p_R, muR) - 1.0;
119 v_R = 2.0 * c_R * v_R / (gammaR-1.0);
120 v_R = u_R + v_R;
121 }
122 gap = fabs(v_L - v_R);
123
124 if (fabs(u_L - u_R) < tol && fabs(p_L - p_R) < tol)
125 {
126 *P_star = 0.5*(p_L + p_R);
127 *U_star = 0.5*(u_L + u_R);
128
129 return fabs(u_L - u_R);
130 }
131
132 //=======THE NEWTON ITERATION=======
133 while((gap > tol) && (n != N))
134 {
135 //the (p,u)-tangent slope on I1 at (v_L,p_INT), i.e. [du/dp](p_INT)
136 if(p_INT > p_L)
137 {
138 delta_p = p_INT - p_L;
139 temp1 = 1.0 / sqrt(1.0 + nuL*delta_p/p_L);
140 temp2 = c_L / gammaL / p_L;
141 temp3 = 0.5 * temp2 * nuL / p_L;
142 k1 = temp3*delta_p*pow(temp1,3.0) - temp2*temp1;
143 }
144 else
145 {
146 temp2 = c_L / gammaL / p_L;
147 temp1 = 1.0 / pow(p_INT/p_L, nuL);
148 k1 = -temp1 * temp2;
149 }
150 //the (p,u)-tangent slope on I3 at (v_R,p_INT), i.e. [du/dp](p_INT)
151 if(p_INT > p_R)
152 {
153 delta_p = p_INT - p_R;
154 temp1 = 1.0 / sqrt(1.0 + nuR*delta_p/p_R);
155 temp2 = c_R / gammaR / p_R;
156 temp3 = 0.5 * temp2 * nuR / p_R;
157 k3 = temp2*temp1 - temp3*delta_p*pow(temp1,3.0);
158 }
159 else
160 {
161 temp2 = c_R / gammaR / p_R;
162 temp1 = 1.0 / pow(p_INT/p_R, nuR);
163 k3 = temp1 * temp2;
164 }
165
166 //the intersect of (u-u_L)=k1*(p-p_L) and (u-u_R)=k3*(p-p_R)
167 p_INT0 = p_INT + (v_R - v_L) / (k1 - k3);
168 if(p_INT0 < 0.0)
169 p_INT = 0.5*p_INT;
170 else
171 p_INT = p_INT0;
172
173 //------the gap------
174 ++n;
175 if(p_INT > p_L)
176 {
177 delta_p = p_INT - p_L;
178 v_L = sqrt(1.0 + nuL*delta_p/p_L);
179 v_L = delta_p * c_L / gammaL / p_L / v_L;
180 v_L = u_L - v_L;
181 }
182 else
183 {
184 v_L = pow(p_INT/p_L, muL) - 1.0;
185 v_L = 2.0 * c_L * v_L / (gammaL-1.0);
186 v_L = u_L - v_L;
187 }
188 if(p_INT > p_R)
189 {
190 delta_p = p_INT - p_R;
191 v_R = sqrt(1.0 + nuR*delta_p/p_R);
192 v_R = delta_p * c_R / gammaR / p_R / v_R;
193 v_R = u_R + v_R;
194 }
195 else
196 {
197 v_R = pow(p_INT/p_R, muR) - 1.0;
198 v_R = 2.0 * c_R * v_R / (gammaR-1.0);
199 v_R = u_R + v_R;
200 }
201
202 gap = fabs(v_L - v_R);
203 }
204
205 u_INT = k1*(v_R-v_L)/(k1-k3)+v_L;
206
207 *P_star = p_INT;
208 *U_star = u_INT;
209
210 return gap;
211}
